# function to convert Request_Closing_Time in Hour

    
def convtoHr(cds):
    days = cds.days
    hours = round(cds.seconds/3600, 2)
    return (days * 24) + hours
  

dt_nys_wrk['Request_Closing_Time_InHr'] = dt_nys_wrk['Request_Closing_Time'].apply(convtoHr)
